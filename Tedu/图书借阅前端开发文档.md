<script>
  export default {
    data() { //该属性中放置用户定义的变量，类似与类的属性定义
      return {
        rules: { //定义表单输入的验证规则
          catename: [{
              required: true,
              message: '请输入类别名称',
              trigger: 'blur'
            }
          ]
        },
        dialogFormEditVisible: false, //显示修改对话框标记
        dialogFormVisible: false, //显示对话框对话框标记
        category: { //添加或者修改时将数据封装成对象
          cateid: 0,
          catename: ''
        },
        pager: { //封装的分页参数
          page: 1, //分页起始数据
          limit: 5, //每页显示最大条数
          total: 0 //统计分页记录数
        },
        cname: '', //定义条件查询接收值的变量
        tableData: [] //定义表格中填充的数据，该数据要从后端服务中获取
      }
    },
    methods: { //该标签中编写的是用户自定义的javascript方法
      openEditDialog(scope) {
        this.category = scope.row; //填充修改对话框表单值
        //填充表单数据
        this.dialogFormEditVisible = !this.dialogFormEditVisible; //打开修改对话框
      },
      update() { //修改方法
       this.$refs['category'].validate(valid=>{
         if(valid){
           this.$http.post('http://127.0.0.1/category/update', this.category)
             .then(res => {
               if (res.data.code == 200) {
                 this.$message({
                   message: res.data.message,
                   type: 'success'
                 })
                 this.findPager();
                 this.dialogFormEditVisible = !this.dialogFormEditVisible //关闭修改对话
               }
             }).catch(err => {
               this.$message.error('错了哦，网络出现异常');
             })
         }

       })

      },
      openAddDialog() { //打开添加对话框
        this.dialogFormVisible = !this.dialogFormVisible //打开或者关闭对话框
      },
      save() {
        this.$refs['category'].validate(valid=>{
          if(valid){
            this.$http.post('http://127.0.0.1/category/save', this.category)
              .then(res => {
                if (res.data.code == 200) {
                  this.$message({
                    message: res.data.message,
                    type: 'success'
                  })
                  this.findPager();
                  this.dialogFormVisible = !this.dialogFormVisible //关闭添加对话
                }
              }).catch(err => {
                this.$message.error('错了哦，网络出现异常');
              })

          }
        })
      },
      remove(scope) {
        this.$http.post('http://127.0.0.1/category/remove?cateid=' + scope.row.cateid)
          .then(res => {
            if (res.data.code == 200) {
              this.$message({
                message: res.data.message,
                type: 'success'
              })
              this.findPager();
            }
          })
          .catch(err => {
            this.$message.error('错了哦，网络出现异常');
          })
      },
      handleSizeChange(val) { //改变每页显示的条数
        console.log(`每页 ${val} 条`);
        this.pager.limit = val
        this.findPager() //再次调用分页查询让数据更新
      },
      handleCurrentChange(val) { //改变当前是第几页
        console.log(`当前页: ${val}`);
        this.pager.page = val;
        this.findPager() //再次调用分页查询让数据更新
      },
      findPager() {
        this.$http.get('http://127.0.0.1/category/findPager', {
          params: {
            page: this.pager.page, //分页起始数据
            limit: this.pager.limit, //每页显示条数
            cname: this.cname //条件查询cname
          }
        }).then(res => {
          console.info(res);
          this.tableData = res.data.data.data;
          this.pager.total = res.data.data.count; //获得分页统计数
          this.pager.page = res.data.data.page;

        }).catch(err => {
          this.$message.error('错了哦，网络出现异常');
        })

      }
    },
    created() {
      this.findPager(); //调用上面的用户自定义方法，使用this关键字即可
    }

  }

            })
            this.findPager();
          }
        })

      openEditDialog(scope) {
        this.category = scope.row; //填充修改对话框表单值
        //填充表单数据
        this.dialogFormEditVisible = !this.dialogFormEditVisible; //打开修改对话框
      },
      update() { //修改方法
        this.$refs['category'].validate(valid => {
          if (valid) {
            this.$http.post('http://127.0.0.1/category/update', this.category)
              .then(res => {
                if (res.data.code == 200) {
                  this.$message({
                    message: res.data.message,
                    type: 'success'
                  })
                  this.findPager();
                  this.dialogFormEditVisible = !this.dialogFormEditVisible //关闭修改对话
                }
